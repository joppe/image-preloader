Image Preloader

var preloader = new m.ImagePreloader(document.getElementsByTagName('img')),
    progressBar = document.querySelector('.progress-bar');

preloader.load({
    progress: function (count, total) {
        progressBar.style.width = ((count / total) * 100) + '%';
    },
    complete: function () {
        console.log('Hell yeah!');
    }
});
